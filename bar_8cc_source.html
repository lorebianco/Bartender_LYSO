56void Bar::SetParsDistro()
57{
58 Int_t status;
59 Double_t my_charge, A, tau_rise, tau_dec;
60
61 // Get a TTree from the text file of parameters
62 TTree *tree = new TTree("tree", "mytree");
63 tree->ReadFile(fInputFilename.c_str());
64
65 tree->SetBranchAddress("Status", &status);
66 tree->SetBranchAddress("my_charge", &my_charge);
67 tree->SetBranchAddress("A", &A);
68 tree->SetBranchAddress("Tau_rise", &tau_rise);
69 tree->SetBranchAddress("Tau_dec", &tau_dec);
70
71 // Create hPars
72 hPars = new TH3D("hPars", "Fitted All", fHisto_A[0], fHisto_A[1], fHisto_A[2], fHisto_Tau_rise[0], fHisto_Tau_rise[1], fHisto_Tau_rise[2], fHisto_Tau_dec[0], fHisto_Tau_dec[1], fHisto_Tau_dec[2]);
73
74 // Fill hPars
75 for(Int_t i = 0; i < tree->GetEntries(); i++)
76 {
77 tree->GetEntry(i);
78
79 // Check conditions about charge
80 if(status==0 && my_charge >= fChargeCuts[0] && my_charge <= fChargeCuts[1])
81 {
82 hPars->Fill(A, tau_rise, tau_dec);
83 }
84 }
85
86 cout << "Distro setted with entries with charge in [" << fChargeCuts[0] << ", " << fChargeCuts[1] << "]" << endl;
87
88 // Delete the TTree
89 delete tree;
90}

94Double_t Bar::Wave_OnePhel(Double_t t, Double_t A, Double_t tau_rise, Double_t tau_dec, Int_t timePhel)
95{
96 Double_t funcVal;
97 Double_t expRise = Exp(-(t-timePhel)/tau_rise);
98 Double_t expDec = Exp(-(t-timePhel)/tau_dec);
99
100 // It happens sometimes when t << timePhel or t >> timePhel, so it's just a numerical fixing
101 if(expRise > DBL_MAX || expRise < DBL_MIN) expRise = 0;
102 if(expDec > DBL_MAX || expDec < DBL_MIN) expDec = 0;
103
104 // 1-Phel function
105 funcVal = (-A*(expDec-expRise))*((t > timePhel) ? 1:0);
106
107 return funcVal;
108}

143void Bar::SetFrontWaveform(Int_t event, Int_t channel, Double_t start)
144{
145 // Sample from hPars the parameters of 1-Phel WF
146 Double_t A, tau_rise, tau_dec;
147 hPars->GetRandom3(A, tau_rise, tau_dec, fRandPars);
148
149 // Evaluate and sum the new 1-Phel WF to the existing one
150 for(Int_t bin = 0; bin < SAMPLINGS; bin++)
151 {
152 fFront[event][channel][bin] += Wave_OnePhel(bin, A, tau_rise, tau_dec, Nint(start+ZERO_TIME_BIN));
153 }

173void Bar::SaveBar()
174{
175 // Determine the output filename based on the Bar ID
176 string outputFilename;
177 if(fID)
178 {
179 outputFilename = "BarID_" + to_string(fID) + ".txt";
180 }
181 else outputFilename = "output.txt";
182
183 // Open the output file for writing, truncating it if it already exists
184 ofstream outputFile(outputFilename, ofstream::out | ofstream::trunc);
185
186 // Check if the output file is successfully opened
187 if(!outputFile.is_open())
188 {
189 cout << "Can't open output file!" << endl;
190 return;
191 }
192
193 // Loop through each event and save data to the output file
194 for(Int_t ev = 0; ev < EVENTS; ev++)
195 {
196 // Write event header
197 outputFile << "#Event " << ev << endl;
198 outputFile << "#" << endl;
199 outputFile << "#Front Detector" << endl;
200 outputFile << "#bin\t";
201
202 // Write channel headers
203 for (Int_t ch = 0; ch < CHANNELS; ch++)
204 {
205 outputFile << "ch" << ch << "\t";
206 }
207 outputFile << "\n";
208
209 // Write Front detector data
210 for(Int_t bin = 0; bin < SAMPLINGS; bin++)
211 {
212 outputFile << bin << "\t";
213 for(Int_t ch = 0; ch < CHANNELS; ch++)
214 {
215 outputFile << fFront[ev][ch][bin] << "\t";
216 }
217 outputFile << endl;
218 }
219
220 // Write a separator and header for the back detector
221 outputFile << "#" << endl;
222 outputFile << "#Back Detector" << endl;
223 outputFile << "#bin\t";
224
225 // Write channel headers for the Back detector
226 for(Int_t ch = 0; ch < CHANNELS; ch++)
227 {
228 outputFile << "ch" << ch << "\t";
229 }
230 outputFile << "\n";
231
232 // Write Back detector data
233 for(Int_t bin = 0; bin < SAMPLINGS; bin++)
234 {
235 outputFile << bin << "\t";
236 for(Int_t ch = 0; ch < CHANNELS; ch++)
237 {
238 outputFile << fBack[ev][ch][bin] << "\t";
239 }
240 outputFile << endl;
241 }
242 }
243
244 // Close the output file
245 outputFile.close();
246
247 // Display a message indicating where the data was saved
248 cout << "\nData saved in " << outputFilename << endl;
249}
