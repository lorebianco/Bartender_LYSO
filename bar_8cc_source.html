60string Bar::GenerateOutputFilename(const char* inputFilename)
61{
62 // Regular expression to extract digits following "MCID_"
63 regex regex("\\bMCID_(\\d+)");
64
65 // Storage for matched results
66 smatch match;
67
68 // Convert inputFilename to string
69 string filename = inputFilename;
70
71 // Check if there is a match for digits following "MCID_" in the filename
72 if(regex_search(filename, match, regex))
73 {
74 // Extract the matched digits as a string
75 string runID = match[1].str();
76
77 if (!runID.empty())
78 {
79 // Convert the extracted string to an integer and assign it to the fID member variable
80 fID = stoi(runID);
81 }
82 }
83
84 // Determine the output filename based on the Bar ID
85 string outputFilename;
86 if (fID)
87 outputFilename = "BarID_" + to_string(fID) + "_t" + to_string(fThreadID) + ".root";
88 else
89 outputFilename = "output.root";
90
91 return outputFilename;
92}

96void Bar::SetParsDistro()
97{
98 Int_t status;
99 Double_t my_charge, A, tau_rise, tau_dec;
100
101 // Get a TTree from the text file of parameters
102 TTree *tree = new TTree("tree", "mytree");
103 tree->ReadFile(fInputFilename.c_str());
104
105 tree->SetBranchAddress("Status", &status);
106 tree->SetBranchAddress("my_charge", &my_charge);
107 tree->SetBranchAddress("A", &A);
108 tree->SetBranchAddress("Tau_rise", &tau_rise);
109 tree->SetBranchAddress("Tau_dec", &tau_dec);
110
111 // Create hPars
112 hPars = new TH3D("hPars", "Fitted All", fHisto_A[0], fHisto_A[1], fHisto_A[2], fHisto_Tau_rise[0], fHisto_Tau_rise[1], fHisto_Tau_rise[2], fHisto_Tau_dec[0], fHisto_Tau_dec[1], fHisto_Tau_dec[2]);
113
114 // Fill hPars
115 for(Int_t i = 0; i < tree->GetEntries(); i++)
116 {
117 tree->GetEntry(i);
118
119 // Check conditions about charge
120 if(status==0 && my_charge >= fChargeCuts[0] && my_charge <= fChargeCuts[1])
121 {
122 hPars->Fill(A, tau_rise, tau_dec);
123 }
124 }
125
126 // Delete the TTree
127 delete tree;
128}

132Double_t Bar::Wave_OnePhel(Double_t t, Double_t A, Double_t tau_rise, Double_t tau_dec, Double_t timePhel)
133{
134 Double_t funcVal;
135 Double_t expRise = Exp(-(t-timePhel)/tau_rise);
136 Double_t expDec = Exp(-(t-timePhel)/tau_dec);
137
138 // It happens sometimes when t << timePhel or t >> timePhel, so it's just a numerical fixing
139 if(expRise > DBL_MAX || expRise < DBL_MIN) expRise = 0;
140 if(expDec > DBL_MAX || expDec < DBL_MIN) expDec = 0;
141
142 // 1-Phel function
143 funcVal = (-A*(expDec-expRise))*((t > timePhel) ? 1:0);
144
145 return funcVal;
146}
